tamano = int(input("Ingrese el tamaño del arreglo: "))
base = int(input("Ingrese el número base: "))

arreglo = [base * i for i in range(1, tamano + 1)]

print(arreglo)

EJERCICIO 2:

tamano = int(input("Ingrese el tamaño de los vectores: "))

nombres = [input(f"Ingrese el nombre {i+1}: ") for i in range(tamano)]
longitudes = [len(nombre) for nombre in nombres]

print("Nombres:", nombres)
print("Longitudes:", longitudes)

EJERCICIO 3:
n = int(input("Ingrese la cantidad de clientes: "))

# Registrar evaluaciones
evaluaciones = [int(input(f"Cliente {i+1}: ")) for i in range(n)]

# a) Contar respuestas
conteo = [evaluaciones.count(i) for i in range(1, 6)]

print("\nRespuestas:")
print("Malo:", conteo[0])
print("Regular:", conteo[1])
print("Buena:", conteo[2])
print("Muy Buena:", conteo[3])
print("Excelente:", conteo[4])

# b) Más frecuente
mas_frecuente = conteo.index(max(conteo)) + 1
print("Más frecuente:", mas_frecuente)

# c) Promedio y porcentaje menor
promedio = sum(evaluaciones) / n
menores = len([x for x in evaluaciones if x < promedio])
porcentaje = (menores / n) * 100

print("Promedio:", round(promedio, 2))
print("Porcentaje menor al promedio:", round(porcentaje, 2), "%")
