def convertir_a_maya(numero):
    if numero == 0:
        return ["○"]

    niveles = []

    while numero > 0:
        residuo = numero % 20
        numero = numero // 20
        niveles.append(residuo)

    representacion = []

    for nivel in niveles:
        if nivel == 0:
            representacion.append("○")
        else:
            barras = nivel // 5
            puntos = nivel % 5
            simbolo = "—" * barras + "●" * puntos
            representacion.append(simbolo)

    return representacion

num = int(input("Ingrese un número: "))

maya = convertir_a_maya(num)

def convertir_a_binario(numero):
    return bin(numero)[2:]

    if opcion == "1":
